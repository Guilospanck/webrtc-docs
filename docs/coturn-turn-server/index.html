Coturn - TURN server

https://www.metered.ca/blog/running-coturn-in-docker-a-step-by-step-guide

realm=my-turn-server-raspi.local
listening-port=3478
tls-listening-port=5349
min-port=49152
max-port=49200
fingerprint
lt-cred-mech
user=turnuser:turnpass
#!/usr/bin/env bash
set -e

run_podman() {
    echo "Running container..."
    command podman run -d --name coturn \
      -p 3478:3478 -p 3478:3478/udp \
      -p 5349:5349 -p 5349:5349/udp \
      -p 49152-49200:49152-49200/udp \
      --mount type=tmpfs,destination=/var/lib/coturn \
      -v "$(pwd)/turnserver.conf:/etc/coturn/turnserver.conf:ro" \
      --replace \
      docker.io/coturn/coturn:4.8.0-alpine
}

configure_ufw() {
    echo "Allowing firewall..."
    sudo ufw allow 3478,5349/tcp comment 'STUN/TURN TCP'
    sudo ufw allow 3478,5349/udp comment 'STUN/TURN UDP'
    sudo ufw allow 49152:49200/udp comment 'TURN relay range'
}

run_podman
configure_ufw

echo "Turn server: turn:192.168.1.118:3478"
